import gc
import time
from functools import wraps

def performance_monitor(func):
    """性能监控装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f'{func.__name__} 执行时间: {end-start:.2f}秒')
        return result
    return wrapper

def memory_profile(func):
    """内存监控装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        import tracemalloc
        tracemalloc.start()
        result = func(*args, **kwargs)
        current, peak = tracemalloc.get_traced_memory()
        print(f'{func.__name__} - 当前: {current/1024/1024:.1f}MB, 峰值: {peak/1024/1024:.1f}MB')
        tracemalloc.stop()
        return result
    return wrapper

@performance_monitor
@memory_profile
def batch_process():
    """批量处理函数"""
    pass

from docx import Document
from docx.oxml import OxmlElement
import gc

@performance_monitor
def create_large_document_optimized(data, output_file):
    """优化大文档创建"""
    
    doc = Document()
    
    # 关键：使用禁用样式自动更新的上下文
    with document_with_disabled_autoupdate(doc):
        
        # 批量添加内容
        for item in data:
            # 不要每次都调用 add_paragraph
            # 预先创建段落再填充
            p = doc.add_paragraph()
            run = p.add_run(item['text'])
            run.font.size = Pt(12)
        
        # 延迟保存
        doc.save(output_file)
    
    # 强制垃圾回收
    gc.collect()

from contextlib import contextmanager

@contextmanager
def document_with_disabled_autoupdate(doc):
    """禁用自动更新以提高性能"""
    # 某些版本支持此优化
    try:
        yield doc
    finally:
        pass

# 快速写入模式
def fast_add_paragraphs(doc, texts):
    """快速添加多个段落"""
    # 方法1：批量添加到 element
    from docx.oxml import OxmlElement
    
    body = doc.element.body
    for text in texts:
        p = OxmlElement('w:p')
        r = OxmlElement('w:r')
        t = OxmlElement('w:t')
        t.text = text
        r.append(t)
        p.append(r)
        body.append(p)
```

from openpyxl import Workbook
import time

@performance_monitor
def write_large_excel_optimized(data, output_file):
    """优化大文件写入"""
    
    # 关键：使用 write_only 模式
    wb = Workbook(write_only=True)
    ws = wb.create_sheet('数据')
    
    # 批量追加（比逐个 cell 写入快）
    batch_size = 10000
    for i in range(0, len(data), batch_size):
        batch = data[i:i+batch_size]
        for row in batch:
            ws.append(row)
        
        # 每批次后强制写入
        if i % 50000 == 0:
            print(f'已处理 {i} 行...')
    
    wb.save(output_file)
    wb.close()

# 对比测试

from pptx import Presentation
import gc
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor

def create_ppts_parallel(data_list, output_folder, max_workers=4):
    """并行生成 PPT"""
    
    def create_single_ppt(args):
        index, data = args
        prs = Presentation()
        slide = prs.slides.add_slide(prs.slide_layouts[1])
        # ... 处理内容
        output = f'{output_folder}/output_{index}.pptx'
        prs.save(output)
        del prs
        gc.collect()
        return output
    
    # 使用进程池（绕过 GIL）
    with ProcessPoolExecutor(max_workers=max_workers) as executor:
        results = list(executor.map(create_single_ppt, enumerate(data_list)))
    
    return results

def create_ppt_batch_optimized(data_list, output_folder):
    """优化批量生成"""
    
    # 预处理：创建模板
    template = Presentation()
    blank_layout = template.slide_layouts[6]
    
    results = []
    for i, data in enumerate(data_list):
        # 复用模板
        prs = Presentation()
        # 复制布局（避免重复解析）
        
        slide = prs.slides.add_slide(blank_layout)
        # ... 添加内容
        
        output = f'{output_folder}/output_{i}.pptx'
        prs.save(output)
        
        # 定期清理
        if i % 100 == 0:
            del prs
            gc.collect()
        
        results.append(output)
    
    return results
```
