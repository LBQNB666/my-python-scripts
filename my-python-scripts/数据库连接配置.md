## 🔗 MySQL 连接

```python
from sqlalchemy import create_engine
import pandas as pd

# 连接 MySQL
engine = create_engine('mysql+pymysql://用户名:密码@localhost:3306/数据库名')

# 读取数据
df = pd.read_sql('SELECT * FROM 表名', engine)

# 写入数据
df.to_sql('表名', engine, if_exists='replace', index=False)

# 执行原生 SQL
with engine.connect() as conn:
    result = conn.execute('SELECT * FROM 表名 LIMIT 10')
    for row in result:
        print(row)
```

### 配置示例
```python
# 本地 MySQL
engine = create_engine('mysql+pymysql://root:123456@localhost:3306/mydb')

# 远程 MySQL
engine = create_engine('mysql+pymysql://用户名:密码@远程IP:3306/数据库名')

# 带连接池
engine = create_engine(
    'mysql+pymysql://用户名:密码@localhost:3306/数据库名',
    pool_size=10,  # 连接池大小
    max_overflow=20  # 超出连接数
)
```

## 🔗 PostgreSQL 连接

```python
from sqlalchemy import create_engine
import pandas as pd

# 连接 PostgreSQL
engine = create_engine('postgresql+psycopg2://用户名:密码@localhost:5432/数据库名')

# 读取数据
df = pd.read_sql('SELECT * FROM 表名', engine)

# 写入数据
df.to_sql('表名', engine, if_exists='replace', index=False)
```

### 配置示例
```python
# 本地 PostgreSQL
engine = create_engine('postgresql+psycopg2://postgres:123456@localhost:5432/mydb')

# 远程 PostgreSQL
engine = create_engine('postgresql+psycopg2://用户名:密码@远程IP:5432/数据库名')

# 带 SSL
engine = create_engine(
    'postgresql+psycopg2://用户名:密码@localhost:5432/数据库名',
    connect_args={'sslmode': 'require'}
)
```

## 🔗 SQLite 连接（最简单，无需安装服务器）

```python
from sqlalchemy import create_engine
import pandas as pd

# 连接 SQLite（直接操作 .db 文件）
engine = create_engine('sqlite:///mydatabase.db')

# 读取数据
df = pd.read_sql('SELECT * FROM 表名', engine)

# 写入数据
df.to_sql('表名', engine, if_exists='replace', index=False)
```

## 📊 完整使用示例

### 例1：从数据库读取数据生成 Excel

```python
from sqlalchemy import create_engine
import pandas as pd
from openpyxl import Workbook

def export_db_to_excel(query, output_file, db_url):
    """从数据库导出数据到 Excel"""
    
    # 连接数据库
    engine = create_engine(db_url)
    
    # 读取数据
    df = pd.read_sql(query, engine)
    
    # 创建 Excel
    wb = Workbook()
    ws = wb.active
    ws.title = '数据'
    
    # 写入表头
    for col, header in enumerate(df.columns, start=1):
        ws.cell(row=1, column=col, value=header)
    
    # 写入数据
    for row_idx, row in enumerate(df.values, start=2):
        for col_idx, value in enumerate(row, start=1):
            ws.cell(row=row_idx, column=col_idx, value=value)
    
    wb.save(output_file)
    print(f'已导出到: {output_file}')
    
    engine.dispose()

# 使用
mysql_url = 'mysql+pymysql://root:密码@localhost:3306/数据库名'
export_db_to_excel(
    'SELECT * FROM 订单 WHERE 日期 > "2026-01-01"',
    '订单报表.xlsx',
    mysql_url
)
```

### 例2：从 Excel 导入数据到数据库

```python
from sqlalchemy import create_engine
import pandas as pd

def import_excel_to_db(excel_file, table_name, db_url):
    """从 Excel 导入数据到数据库"""
    
    # 读取 Excel
    df = pd.read_excel(excel_file)
    
    # 连接数据库
    engine = create_engine(db_url)
    
    # 写入数据库
    df.to_sql(table_name, engine, if_exists='replace', index=False)
    
    print(f'已导入 {len(df)} 行到 {table_name}')
    
    engine.dispose()

# 使用
import_excel_to_db('员工数据.xlsx', '员工', mysql_url)
```

### 例3：数据库表同步

```python
from sqlalchemy import create_engine
import pandas as pd

def sync_tables(source_db_url, target_db_url, table_name):
    """同步两个数据库的表"""
    
    source_engine = create_engine(source_db_url)
    target_engine = create_engine(target_db_url)
    
    # 从源数据库读取
    df = pd.read_sql(f'SELECT * FROM {table_name}', source_engine)
    
    # 写入目标数据库
    df.to_sql(table_name, target_engine, if_exists='replace', index=False)
    
    print(f'已同步 {len(df)} 行到目标数据库')
    
    source_engine.dispose()
    target_engine.dispose()

# 使用：本地数据库同步到远程
sync_tables(
    'mysql+pymysql://root:密码@localhost:3306/local_db',
    'mysql+pymysql://root:密码@远程IP:3306/remote_db',
    '订单表'
)
```

## 🔧 数据库配置信息

你可以在下面填写你的数据库配置：

```python
# MySQL 配置
MYSQL_CONFIG = {
    'host': 'localhost',
    'port': 3306,
    'user': 'your_username',
    'password': 'your_password',
    'database': 'your_database'
}

# PostgreSQL 配置
POSTGRES_CONFIG = {
    'host': 'localhost',
    'port': 5432,
    'user': 'your_username',
    'password': 'your_password',
    'database': 'your_database'
}

# SQLite 配置（最简单）
SQLITE_PATH = 'path/to/your/database.db'
```

## ⚠️ 注意事项

1. **安全性**：不要把密码直接写在代码里，使用环境变量
2. **性能**：大量数据时使用 `chunksize` 分批处理
3. **连接管理**：使用完记得 `engine.dispose()`
4. **备份**：操作前先备份数据库
